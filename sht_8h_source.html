62/**
63 * Hash function type.
64 *
65 * Callback function type used to calculate hash values for keys.
66 *
67 * 32-bit example, using
68 * [`XXH32()`](https://xxhash.com/doc/v0.8.3/group___x_x_h32__family.html),
69 * which requires a seed:
70 *
71 * ```c
72 * uint32_t myhash32(const void *restrict key, void *restrict context)
73 * {
74 * const struct mykey *k = key;
75 * const XXH32_hash_t *seed = ctx;
76 *
77 * return XXH32(k, sizeof *k, *seed);
78 * }
79 * ```
80 *
81 * 64-bit example, using
82 * [`XXH3_64bits()`](https://xxhash.com/doc/v0.8.3/group___x_x_h3__family.html),
83 * which does not require a seed:
84 *
85 * ```c
86 * uint32_t myhash64(const void *restrict key, const void *restrict)
87 * {
88 * const struct mykey *k = key;
89 *
90 * // Throw away the upper 32 bits
91 * return (uint32_t)XXH3_64bits(k, sizeof *k);
92 * }
93 * ```
94 *
95 * @param key The key to be hashed.
96 * @param context Optional function-specific context.
97 *
98 * @returns The hash value of the key.
99 */
100typedef uint32_t (*sht_hashfn_t)(const void *restrict key,
101 void *restrict context);
102
103/**
104 * Equality comparison function type.
105 *
106 * Callback function type used to compare a key with the key of an existing
107 * bucket. This function is only called when the lower 24 bits of the hash
108 * values of the two keys are equal.
109 *
110 * ```c
111 * struct my_entry {
112 * const char *name;
113 * struct in_addr address;
114 * };
115 *
116 * bool my_cmp(const void *restrict key, const void *restrict entry,
117 * void *restrict)
118 * {
119 * const char *const name = key;
120 * const struct my_entry *const e = entry;
121 *
122 * return strcmp(name, e->name) == 0;
123 * }
124 * ```
125 *
126 * @param key The key to be compared against the key in the
127 * bucket.
128 * @param entry The entry whose key is to be compared.
129 * @param context Optional function-specific context.
130 *
131 * @returns A boolean value that indicates whether @p key is equal to the
132 * key in @p entry.
133 */
134typedef bool (*sht_eqfn_t)(const void *restrict key,
135 const void *restrict entry,
136 void *restrict context);
137
138/**
139 * Free function type.
140 *
141 * Callback function type used to free entry resources. For example:
142 *
143 * ```c
144 * struct my_entry {
145 * const char *name;
146 * struct in_addr address;
147 * };
148 *
149 * void my_free(const void *restrict entry, void *restrict)
150 * {
151 * const struct my_entry *const e = entry;
152 * free(e->name);
153 * }
154 * ```
155 *
156 * @param entry The entry whose resources should be freed.
157 * @param context Optional function-specific context.
158 */
159typedef void (*sht_freefn_t)(const void *restrict entry,
160 void *restrict context);

363/**
364 * @internal
365 * @brief
366 * Macro that expands to its second argument.
367 *
368 * This macro is used by SHT_HT_NEW() to accept an "optional" `err` argument.
369 *
370 * @param _ Dummy argument; only present to make syntax valid.
371 * @param a2 The expression to which this macro will expand.
372 * @param ... Possible additional arguments. Not used.
373 *
374 * @returns This macro expands to its second argument (@p a2).
375 */
376#define SHT_ARG2(_, a2, ...) (a2)
377
378/**
379 * Create a new hash table.
380 *
381 * This macro is a wrapper for sht_new_().
382 *
383 * ```c
384 * struct entry {
385 * const char *name;
386 * struct in_addr address;
387 * };
388 *
389 * struct sht_ht *ht;
390 * enum sht_err err;
391 *
392 * ht = sht_new(hashfn, eqfn, NULL, sizeof(struct entry),
393 * alignof(struct entry), &err);
394 *
395 * // Rewrite as ...
396 * ht = SHT_NEW(hashfn, eqfn, NULL, struct entry, &err);
397 *
398 * // Without error reporting ...
399 * ht = sht_new(hashfn, eqfn, NULL, sizeof(struct entry),
400 * alignof(struct entry), NULL);
401 *
402 * // Becomes ...
403 * ht = SHT_NEW(hashfn, eqfn, NULL, struct entry);
404 * ```
405 *
406 * @param hashfn Function to be used to compute the hash values of keys.
407 * @param eqfn Function to be used to compare keys for equality.
408 * @param freefn Function to be used to free entry resources. (May be
409 * `NULL`.)
410 * @param etype The type of the entries to be stored in the table.
411 * @param[out] ... Optional output pointer for error reporting.
412 *
413 * @returns On success, a pointer to the new hash table is returned. On
414 * error, `NULL` is returned, and an error code is returned in
415 * @p err (if it is not `NULL`).
416 *
417 * @see sht_new_()
418 */
419#define SHT_NEW(hashfn, eqfn, freefn, etype, ...) \
420 ({ \
421 static_assert(sizeof(etype) <= SHT_MAX_ESIZE, \
422 "Entry type (" #etype ") too large"); \
423 sht_new_(hashfn, eqfn, freefn, \
424 sizeof(etype), alignof(etype), \
425 SHT_ARG2(_, ##__VA_ARGS__, NULL)); \
426 })
