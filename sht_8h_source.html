34/**
35 * Critical error printing function.
36 *
37 * If the calling program violates this library's contract, the library will
38 * print an error message and abort the program. This variable can be set to
39 * customize how the error message is printed or logged.
40 *
41 * ```c
42 * static void log_sht_err(const char *msg)
43 * {
44 * syslog(LOG_CRIT, "SHT library error: %s", msg);
45 * }
46 *
47 * sht_abort_print = log_sht_err;
48 * ```
49 *
50 * The library will abort the program if any of the following occur.
51 *
52 * * An invalid value is passed to sht_msg().
53 *
54 * * A `NULL` function pointer is passed to sht_new_().
55 *
56 * * An invalid load factor threshold is passed to sht_set_lft().
57 *
58 * * One of the functions in the table below is called on a table that is in an
59 * inappropriate state.
60 *
61 * | Function | Uninitialized | Initialized | Iterator(s) exist |
62 * |----------------------|:-------------:|:-----------:|:-----------------:|
63 * |sht_set_hash_ctx() | | **ABORT** | † |
64 * |sht_set_eq_ctx() | | **ABORT** | † |
65 * |sht_set_freefn() | | **ABORT** | † |
66 * |sht_set_lft() | | **ABORT** | † |
67 * |sht_init() | | **ABORT** | † |
68 * |sht_free() | | | **ABORT** |
69 * |sht_add() | **ABORT** | | **ABORT** |
70 * |sht_set() | **ABORT** | | **ABORT** |
71 * |sht_get() | **ABORT** | | |
72 * |sht_size() | **ABORT** | | |
73 * |sht_empty() | **ABORT** | | |
74 * |sht_delete() | **ABORT** | | **ABORT** |
75 * |sht_pop() | **ABORT** | | **ABORT** |
76 * |sht_replace() | **ABORT** | | |
77 * |sht_swap() | **ABORT** | | |
78 * |sht_ro_iter() | **ABORT** | | |
79 * |sht_rw_iter() | **ABORT** | | |
80 *
81 * † Abort implied. (An iterator cannot be created on an uninitialized
82 * table.)
83 *
84 * @param msg The error message (not newline terminated).
85 */
86extern void (*sht_abort_print)(const char *msg);

97/**
98 * Hash function type.
99 *
100 * Callback function type used to calculate hash values for keys.
101 *
102 * 32-bit example, using
103 * [`XXH32()`](https://xxhash.com/doc/v0.8.3/group___x_x_h32__family.html),
104 * which requires a seed:
105 *
106 * ```c
107 * uint32_t myhash32(const void *restrict key, void *restrict context)
108 * {
109 * const struct mykey *k = key;
110 * const XXH32_hash_t *seed = ctx;
111 *
112 * return XXH32(k, sizeof *k, *seed);
113 * }
114 * ```
115 *
116 * 64-bit example, using
117 * [`XXH3_64bits()`](https://xxhash.com/doc/v0.8.3/group___x_x_h3__family.html),
118 * which does not require a seed:
119 *
120 * ```c
121 * uint32_t myhash64(const void *restrict key, const void *restrict)
122 * {
123 * const struct mykey *k = key;
124 *
125 * // Throw away the upper 32 bits
126 * return (uint32_t)XXH3_64bits(k, sizeof *k);
127 * }
128 * ```
129 *
130 * @param key The key to be hashed.
131 * @param context Optional function-specific context.
132 *
133 * @returns The hash value of the key.
134 */
135typedef uint32_t (*sht_hashfn_t)(const void *restrict key,
136 void *restrict context);
137
138/**
139 * Equality comparison function type.
140 *
141 * Callback function type used to compare a key with the key of an existing
142 * bucket. This function is only called when the full hash values of the two
143 * keys are equal.
144 *
145 * ```c
146 * struct my_entry {
147 * const char *name;
148 * struct in_addr address;
149 * };
150 *
151 * _Bool my_cmp(const void *restrict key, const void *restrict entry,
152 * void *restrict)
153 * {
154 * const char *const name = key;
155 * const struct my_entry *const e = entry;
156 *
157 * return strcmp(name, e->name) == 0;
158 * }
159 * ```
160 *
161 * @param key The key to be compared against the key in the
162 * bucket.
163 * @param entry The entry whose key is to be compared.
164 * @param context Optional function-specific context.
165 *
166 * @returns A boolean value that indicates whether @p key is equal to the
167 * key in @p entry.
168 */
169typedef _Bool (*sht_eqfn_t)(const void *restrict key,
170 const void *restrict entry,
171 void *restrict context);
172
173/**
174 * Free function type.
175 *
176 * Callback function type used to free entry resources.. For example:
177 *
178 * ```c
179 * struct my_entry {
180 * const char *name;
181 * struct in_addr address;
182 * };
183 *
184 * void my_free(const void *restrict entry, void *restrict)
185 * {
186 * const struct my_entry *const e = entry;
187 * free(e->name);
188 * }
189 * ```
190 *
191 * @param entry The entry whose resources should be freed.
192 * @param context Optional function-specific context.
193 */
194typedef void (*sht_freefn_t)(const void *restrict entry,
195 void *restrict context);

420
421/**
422 * @internal
423 * Macro that expands to its second argument.
424 *
425 * This macro is used by SHT_HT_NEW() to accept an "optional" `err` argument.
426 *
427 * @param _ Dummy argument; only present to make syntax valid.
428 * @param a2 The expression to which this macro will expand.
429 * @param ... Possible additional arguments. Not used.
430 *
431 * @returns This macro expands to its second argument (@p a2).
432 */
433#define SHT_ARG2(_, a2, ...) (a2)
434
435/**
436 * Create a new hash table.
437 *
438 * This macro is a wrapper for sht_new_().
439 *
440 * ```c
441 * struct entry {
442 * const char *name;
443 * struct in_addr address;
444 * };
445 *
446 * struct sht_ht *ht;
447 * enum sht_err err;
448 *
449 * ht = sht_new(hashfn, eqfn, sizeof(struct entry),
450 * _Alignof(struct entry), &err);
451 *
452 * // Rewrite as ...
453 * ht = SHT_NEW(hashfn, eqfn, struct entry, &err);
454 *
455 * // Without error reporting ...
456 * ht = sht_new(hashfn, eqfn, sizeof(struct entry),
457 * _Alignof(struct entry), NULL);
458 *
459 * // Becomes ...
460 * ht = SHT_NEW(hashfn, eqfn, struct entry);
461 * ```
462 *
463 * @param hashfn The function that will be used to compute the hash
464 * values of keys.
465 * @param eqfn The function that will be used to compare keys for
466 * equality.
467 * @param etype The type of the entries to be stored in the table.
468 * @param[out] ... Optional output pointer for error reporting.
469 *
470 * @returns On success, a pointer to the new hash table is returned. On
471 * error, `NULL` is returned, and an error code is returned in
472 * @p err (if it is not `NULL`).
473 *
474 * @see sht_new_()
475 */
476#define SHT_NEW(hashfn, eqfn, etype, ...) \
477 sht_new_(hashfn, eqfn, sizeof(etype), _Alignof(etype), \
478 SHT_ARG2(_, ##__VA_ARGS__, NULL))

485/**
486 * @internal
487 * Template macro for generic iterator helpers.
488 *
489 * @param op The "operation" portion of the underlying function
490 * names. E.g. `next` generates a helper macro for
491 * sht_ro_iter_next_() and sht_rw_iter_next_().
492 * @param iter The iterator that will be passed to the underlying
493 * function.
494 * @param ... Any additional arguments that will be passed to the
495 * underlying function.
496 *
497 * @returns This macro expands to a generic helper macro that calls the
498 * correct underlying function for read-only or read/write
499 * iterators.
500 */
501#define SHT_ITER_GENERIC(op, iter, ...) \
502 _Generic((iter), \
503 struct sht_ro_iter *: sht_ro_iter_##op##_, \
504 struct sht_rw_iter *: sht_rw_iter_##op##_ \
505 )(iter, ##__VA_ARGS__)

513 * @see sht_rw_iter_free_()
514 */
515#define SHT_ITER_FREE(iter) SHT_ITER_GENERIC(free, iter)

524 *
